# 4-scriptin
# for bash script
touch sript.sh # creates a file
nano script.sh # edit file in text editor
chmod +x script.sh # adds execution status to file
./script.sh variable_name # runs file with the given variable
# Bash script is executed by interpretation

# for python script
touch sript.py # creates a file
nano script.py # edit file in text editor
chmod +x script.py # adds execution status to file
python script.py variable_name # runs file with the given variable
# Python script is executed using a hybrid approach that combines both compilation and interpretation
